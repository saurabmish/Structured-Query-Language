CREATE TABLE employees (
  employee_id SERIAL PRIMARY KEY,
  first_name varchar(50) NOT NULL,
  last_name varchar(50) NOT NULL,
  job_title varchar(50) NOT NULL,
  salary integer NOT NULL,
  reports_to integer DEFAULT NULL,
  office_id integer NOT NULL,
  fk_office_id integer,
  fk_reports_to integer,
  CONSTRAINT fk_employees_managers FOREIGN KEY (fk_reports_to) REFERENCES employees (employee_id),
  CONSTRAINT fk_employees_offices FOREIGN KEY (fk_office_id) REFERENCES offices (office_id) ON UPDATE CASCADE
);
